# Python Code for Multi-AI Tool Integration

```python
import requests
import json

# Sample APIs (Example URLs)
openai_api = "https://api.openai.com/v1/chat/completions"
gemini_api = "https://api.gemini.com/generate"

# API Keys
OPENAI_KEY = "your_openai_api_key"
GEMINI_KEY = "your_gemini_api_key"

# Prompt
prompt = "Write a Python function to check whether a number is prime."

# Function to call OpenAI API
def get_openai_response(prompt):
    headers = {
        "Authorization": f"Bearer {OPENAI_KEY}",
        "Content-Type": "application/json"
    }

    data = {
        "model": "gpt-4",
        "messages": [{"role": "user", "content": prompt}]
    }

    response = requests.post(openai_api, headers=headers, json=data)

    return response.json()

# Function to call Gemini API
def get_gemini_response(prompt):
    headers = {
        "Authorization": f"Bearer {GEMINI_KEY}",
        "Content-Type": "application/json"
    }

    data = {
        "prompt": prompt
    }

    response = requests.post(gemini_api, headers=headers, json=data)

    return response.json()

# Collect responses
openai_output = get_openai_response(prompt)
gemini_output = get_gemini_response(prompt)

# Display outputs
print("OpenAI Response:")
print(json.dumps(openai_output, indent=2))

print("\nGemini Response:")
print(json.dumps(gemini_output, indent=2))
```

def is_prime(n):
    if n <= 1:
        return False

    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False

    return True
